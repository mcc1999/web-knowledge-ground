## 图片 alt 属性占位，动态赋值 src 属性

- `img` 标签不设置 src (不是 src="", 应不写 src)表现为普通内联元素，而不是可替换元素，Chrome 需设置 alt 属性才是以上表现
- 可替换元素没有伪类元素

<Sandpack 
  template="static"
  theme={aquaBlue}
  options={{
    showLineNumbers: true, // default - true
    showInlineErrors: true, // default - false
    wrapContent: true, // default - false
    editorHeight: 480, // default - 300
    editorWidthPercentage: 60, // default - 50
  }}
  files={{
    "/index.html": {
      code: 
        `<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <meta http-equiv="X-UA-Compatible" content="ie=edge" />
    <title>Static Template</title>
    <style>
      img {
        width: 256px;
        height: 192px;
        display: inline-block;
        color: transparent;
        position: relative;
        overflow: hidden;
      }
      img:not([src]) {
        visibility: hidden;
      }
      /* 可替换元素没有伪类元素 */
      img::before {
        content: '';
        width: 100%;
        height: 100%;
        position: absolute;
        left: 0;
        background: #f0f3f9;
        visibility: visible;
      }
      img::after {
        content: attr(alt);
        width: 100%;
        height: 24px;
        text-align: center;
        line-height: 24px;
        position: absolute;
        bottom: 0;
        left:0;
        visibility: visible;
        color: white;
        background: rgba(0, 0, 0, 0.5);
        transform: translateY(100%);
        transition: tranform 0.2s;
      }
      img:hover::after {
        transform: translateY(0);
      }
      #btn {
        display: block;
      }
    </style>

  </head>
  <body>
    <img id="img" alt="IMG ALT占位符" />
    <button id="btn" onclick="setSrc()">设置src</button>
    <script>
      let hasSrc = false
      const imgEle = document.getElementById('img')
      const btnEle = document.getElementById('btn')
      function setSrc() {
        if (!hasSrc) {
          imgEle.setAttribute('src', 'https://demo.cssworld.cn/images/common/l/1.jpg')
          btnEle.innerHTML = '移除src'
          hasSrc = true
        } else {
          imgEle.removeAttribute('src')
          btnEle.innerHTML = '设置src'
          hasSrc = false
        }
      }
    </script>
  </body>
</html>
`
      }

}}
/>
